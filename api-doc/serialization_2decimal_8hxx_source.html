00021         // The formula for the number of decimla digits required is given in:
00022         //
00023         // http://www.open-std.org/JTC1/SC22/WG21/docs/papers/2005/n1822.pdf
00024         //
00025         template <typename C>
00026         std::basic_string<C>
00027         insert (const as_decimal<double>& d)
00028         {
00029           std::basic_ostringstream<C> os;
00030           os.imbue (std::locale::classic ());
00031           std::streamsize prec;
00032 
00033           const facet* f = d.facets ?
00034             facet::find (d.facets, facet::fraction_digits) : 0;
00035 
00036           if (f)
00037             prec = static_cast<std::streamsize> (f->value);
00038           else
00039           {
00040             // Precision.
00041             //
00042 #if defined (XSD_CXX_TREE_DECIMAL_PRECISION_MAX)
00043             prec = 2 + std::numeric_limits<double>::digits * 301/1000;
00044 #elif defined (XSD_CXX_TREE_DECIMAL_PRECISION)
00045             prec = XSD_CXX_TREE_DECIMAL_PRECISION;
00046 #else
00047             prec = std::numeric_limits<double>::digits10;
00048 #endif
00049           }
00050 
00051           os.precision (prec);
00052           os << std::fixed << d.x;
00053           std::basic_string<C> r (os.str ());
00054           const C* cr (r.c_str ());
00055 
00056           // Remove the trailing zeros and the decimal point if necessary.
00057           //
00058           typename std::basic_string<C>::size_type size (r.size ()), n (size);
00059 
00060           if (prec != 0)
00061           {
00062             for (; n > 0 && cr[n - 1] == '0'; --n)/*noop*/;
00063 
00064             if (n > 0 && cr[n - 1] == '.')
00065               --n;
00066           }
00067 
00068           // See if we have a restriction on total digits.
00069           //
00070           f = d.facets ? facet::find (d.facets, facet::total_digits) : 0;
00071 
00072           if (f && n > f->value)
00073           {
00074             // Point and sign do not count so figure out if we have them.
00075             //
00076             typename std::basic_string<C>::size_type extra (
00077               cr[0] == '-' ? 1 : 0);
00078 
00079             if (r.find ('.') < n)
00080               extra++;
00081 
00082             // Unless we have a point and the size difference is one,
00083             // remove some digits.
00084             //
00085             if ((n - extra) > f->value)
00086               n -= (n - extra - f->value);
00087 
00088             if (n > 0 && cr[n - 1] == '.')
00089               --n;
00090           }
00091 
00092           if (n != size)
00093             r.resize (n);
00094 
00095           return r;
00096         }
00097       }
