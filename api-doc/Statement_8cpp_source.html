00070 {

00075     int status = sqlite3_step(stmt);
00076     if (status == SQLITE_ROW)

00082     if (status == SQLITE_DONE)

00084         reset_internal();

00086         dn = true;
00087         return;

00089     if (status == SQLITE_BUSY)
00090     {
00091         std::string msg(sqlite3_errmsg(db->cobj()));
00092         reset_nothrow();
00093         throw SqliteException(status, msg);
00094     }
00095     if (status == SQLITE_MISUSE)
00096     {
00097         reset_nothrow();
00098         throw_sqlite_status(status, db->cobj());
00099     }
