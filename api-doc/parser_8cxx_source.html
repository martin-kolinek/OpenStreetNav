00156 xml::dom::auto_ptr<DOMDocument> parser_impl::
00157 next ()
00158 {
00159   // We should be at depth 1. If not, then we are done parsing.
00160   //
00161   if (depth_ != 1)
00162     return xml::dom::auto_ptr<DOMDocument> (0);
00163 
00164   bool r (true);
00165 
00166   // Keep calling parseNext() until we either move to a greater depth or
00167   // get a document. This way we skip the text (presumably whitespaces)
00168   // that may be preceding the next chunk.
00169   //
00170   while (r && depth_ == 1 && doc_.get () == 0)
00171   {
00172     parser_->parseNext (token_);
00173     error_handler_.throw_if_failed<tree::parsing<char> > ();
00174   }
00175 
00176   if (!r)
00177     return xml::dom::auto_ptr<DOMDocument> (0);
00178 
00179   // If we are not at depth 1, keep calling parseNext() until we get
00180   // there.
00181   //
00182   while (r && depth_ != 1)
00183   {
00184     r = parser_->parseNext (token_);
00185     error_handler_.throw_if_failed<tree::parsing<char> > ();
00186   }
